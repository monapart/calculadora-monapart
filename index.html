    <script>

        const inputs = document.querySelectorAll('input');

       

        function calcular() {

            // Valores de entrada

            const v = parseInt(document.getElementById('ventas').value);

            const f = parseInt(document.getElementById('fees').value);

            const s = parseInt(document.getElementById('split').value) / 100;

            document.getElementById('val-ventas').innerText = v;

            document.getElementById('val-fees').innerText = f.toLocaleString();

            document.getElementById('val-split').innerText = (s * 100).toFixed(0);

            // 1. CALCULO ACTUAL

            const facturacionActual = v * f;

            const netoActual = facturacionActual * s;

            // 2. CALCULO MONAPART

            const ventasMona = v * 1.2; // +20% ventas

            const feesMona = f * 1.1;   // +10% fees

            const totalFacturadoMona = ventasMona * feesMona;

            // Split progresivo

            let netoRetenidoMona = 0;

            if (totalFacturadoMona <= 40000) {

                netoRetenidoMona = totalFacturadoMona * 0.73;

            } else if (totalFacturadoMona <= 80000) {

                netoRetenidoMona = (40000 * 0.73) + (totalFacturadoMona - 40000) * 0.77;

            } else if (totalFacturadoMona <= 120000) {

                netoRetenidoMona = (40000 * 0.73) + (40000 * 0.77) + (totalFacturadoMona - 80000) * 0.88;

            } else {

                netoRetenidoMona = (40000 * 0.73) + (40000 * 0.77) + (40000 * 0.88) + (totalFacturadoMona - 120000) * 0.99;

            }

            // Gastos Monapart

            const gastosFijosMona = 600 + (ventasMona * 100);

            const netoFinalMona = netoRetenidoMona - gastosFijosMona;

            // Pintar resultados

            document.getElementById('neto-actual').innerText = Math.round(netoActual).toLocaleString() + '€';

            document.getElementById('neto-monapart').innerText = Math.round(netoFinalMona).toLocaleString() + '€';

           

            const diferencia = netoFinalMona - netoActual;

            document.getElementById('ahorro-total').innerText = '+' + Math.round(diferencia).toLocaleString() + '€ de beneficio extra';

        }

        inputs.forEach(input => input.addEventListener('input', calcular));

        calcular();

    </script>
